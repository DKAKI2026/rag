from typing import Annotated
from pydantic import Field
from azure.search.documents import SearchClient
from backend.config.settings import SEARCH_ENDPOINT, EMBEDDING_MODEL, TOP_K_RESULTS

def get_embedding(embedding_client, texte: str) -> list:
    """Transforme un texte en vecteur numerique"""
    return embedding_client.embeddings.create(
        input=texte, model=EMBEDDING_MODEL
    ).data[0].embedding

def creer_rag_tool(credential, embedding_client, index_name: str):
    """
    Cree un tool RAG pour un index Azure AI Search donne.
    Retourne une fonction decoree @tool prete pour agent_framework.
    """
    from agent_framework import tool

    @tool
    def chercher_documents(
        question: Annotated[str, Field(description="La question ou sujet a rechercher dans les documents")]
    ) -> str:
        """Recherche les documents pertinents dans Azure AI Search."""
        try:
            q_embedding   = get_embedding(embedding_client, question)
            search_client = SearchClient(SEARCH_ENDPOINT, index_name, credential)
            resultats     = list(search_client.search(
                search_text=question,
                vector_queries=[{"vector": q_embedding, "fields": "embedding", "k": TOP_K_RESULTS, "kind": "vector"}]
            ))
            if not resultats:
                return "Aucun document pertinent trouve."
            return "\n\n---\n\n".join([
                f"[Source: {r['fichier']} - page {r['page']}]\n{r['contenu']}"
                for r in resultats
            ])
        except Exception as e:
            return f"Erreur de recherche : {str(e)}"

    return chercher_documents

import os
from typing import Annotated
from datetime import datetime
from pydantic import Field
from agent_framework import tool
from backend.config.settings import (
    DOSSIER_STOCKS, COLONNES_STOCKS,
    SEUIL_STOCK_BAS, SEUIL_STOCK_CRITIQUE
)

@tool
def lire_stocks(
    filtre_modele: Annotated[str, Field(description="Filtrer par modele de voiture (vide = tout)")] = ""
) -> str:
    """Lit le fichier Excel des stocks et retourne l'inventaire actuel."""
    try:
        import pandas as pd
        fichier = f"{DOSSIER_STOCKS}/stocks.xlsx"
        if not os.path.exists(fichier):
            return f"Fichier stocks introuvable : {fichier}"
        df  = pd.read_excel(fichier)
        col = COLONNES_STOCKS
        if filtre_modele:
            df = df[df[col["modele_voiture"]].str.contains(filtre_modele, case=False, na=False)]
        resultats = []
        for _, row in df.iterrows():
            try:
                quantite = int(row[col["quantite"]])
                seuil    = int(row[col["seuil_minimum"]])
                statut   = "OK" if quantite >= seuil else ("CRITIQUE" if quantite < SEUIL_STOCK_CRITIQUE else "BAS")
                resultats.append(f"{row[col['nom']]} | Qte: {quantite} | Seuil: {seuil} | {statut} | Fournisseur: {row.get(col['fournisseur'], 'N/A')}")
            except Exception:
                continue
        return "\n".join(resultats) if resultats else "Aucun stock trouve."
    except Exception as e:
        return f"Erreur lecture stocks : {str(e)}"

@tool
def detecter_alertes_stocks() -> str:
    """Analyse le stock et retourne toutes les alertes : critique, bas, retards."""
    try:
        import pandas as pd
        fichier = f"{DOSSIER_STOCKS}/stocks.xlsx"
        if not os.path.exists(fichier):
            return f"Fichier stocks introuvable : {fichier}"
        df      = pd.read_excel(fichier)
        col     = COLONNES_STOCKS
        alertes = []
        for _, row in df.iterrows():
            try:
                quantite    = int(row[col["quantite"]])
                nom_piece   = row[col["nom"]]
                fournisseur = row.get(col["fournisseur"], "N/A")
                if quantite < SEUIL_STOCK_CRITIQUE:
                    alertes.append(f"CRITIQUE : {nom_piece} — {quantite} unites — Fournisseur: {fournisseur}")
                elif quantite < SEUIL_STOCK_BAS:
                    alertes.append(f"ATTENTION : {nom_piece} — {quantite} unites — Fournisseur: {fournisseur}")
                date_livraison = row.get(col["date_livraison"])
                if pd.notna(date_livraison):
                    if isinstance(date_livraison, str):
                        date_livraison = datetime.strptime(date_livraison, "%Y-%m-%d")
                    if hasattr(date_livraison, 'date'):
                        date_livraison = datetime.combine(date_livraison, datetime.min.time())
                    if date_livraison < datetime.now() and quantite < int(row[col["seuil_minimum"]]):
                        alertes.append(f"RETARD : {nom_piece} — Livraison prevue le {date_livraison.strftime('%d/%m/%Y')}")
            except Exception:
                continue
        if not alertes:
            return "Aucune alerte de stock detectee."
        return f"{len(alertes)} alerte(s) :\n" + "\n".join(alertes)
    except Exception as e:
        return f"Erreur analyse stocks : {str(e)}"

@tool
def chercher_commande(
    recherche: Annotated[str, Field(description="Numero de commande ou nom du client")]
) -> str:
    """Cherche une commande par numero ou par nom de client."""
    try:
        import pandas as pd
        fichier = f"{DOSSIER_COMMANDES}/commandes.xlsx"

            resultats.append(
                f"Commande #{row[col['commande_id']]}\n"

                f"  Voiture : {row[col['voiture_modele']]}\n"
                f"  Statut  : {row[col['statut']]}\n"
                f"  Montant : {row[col['montant']]}$\n"
                f"  Vendeur : {row[col['vendeur']]}\n"
                f"  Date    : {row[col['date_commande']]}"

@tool
def identifier_clients_a_risque(
    jours_inactif: Annotated[int, Field(description="Nombre de jours sans visite pour considerer un client a risque")] = 90
) -> str:
    """Identifie les clients inactifs depuis X jours."""
    try:
        import pandas as pd
        from datetime import datetime, timedelta
        fichier = f"{DOSSIER_CLIENTS}/clients.xlsx"
        if not os.path.exists(fichier):
            return f"Fichier clients introuvable : {fichier}"
        df      = pd.read_excel(fichier)
        col     = COLONNES_CLIENTS
        limite  = datetime.now() - timedelta(days=jours_inactif)
        risques = []
        for _, row in df.iterrows():
            try:
                derniere_visite = pd.to_datetime(row[col["derniere_visite"]])
                if derniere_visite < limite:
                    jours = (datetime.now() - derniere_visite.to_pydatetime()).days
                    risques.append(f"{row[col['nom']]} ({row[col['niveau_fidelite']]}) — Inactif depuis {jours} jours | Email: {row[col['email']]}")
            except Exception:
                continue
        if not risques:
            return f"Aucun client inactif depuis plus de {jours_inactif} jours."
        return f"{len(risques)} client(s) a risque :\n" + "\n".join(risques)
    except Exception as e:
        return f"Erreur analyse clients : {str(e)}"

@tool
def recommander_action_fidelisation(
    client_nom: Annotated[str, Field(description="Nom du client pour lequel generer une recommandation")]
) -> str:
    """Analyse le profil d'un client et suggere une action de fidelisation."""
    try:
        import pandas as pd
        from datetime import datetime
        fichier = f"{DOSSIER_CLIENTS}/clients.xlsx"
        if not os.path.exists(fichier):
            return f"Fichier clients introuvable : {fichier}"
        df  = pd.read_excel(fichier)
        col = COLONNES_CLIENTS
        df  = df[df[col["nom"]].str.contains(client_nom, case=False, na=False)]
        if df.empty:
            return f"Client '{client_nom}' introuvable."
        row             = df.iloc[0]
        achats          = int(row[col["achats_total"]])
        valeur          = float(row[col["valeur_total"]])
        niveau          = row[col["niveau_fidelite"]]
        derniere_visite = pd.to_datetime(row[col["derniere_visite"]])
        jours_inactif   = (datetime.now() - derniere_visite.to_pydatetime()).days
        actions = []
        if jours_inactif > 90:
            actions.append(f"Envoyer email de relance — inactif depuis {jours_inactif} jours")
        if achats >= 3 and niveau == "Bronze":
            actions.append("Proposer montee en niveau Argent")
        if valeur > 50000:
            actions.append("Inviter a un evenement VIP exclusif")
        if achats == 1:
            actions.append("Offrir remise de 5% sur le prochain achat")
        if not actions:
            actions.append("Client actif — maintenir contact mensuel")
        return (
            f"Client: {row[col['nom']]} ({niveau})\n"
            f"Valeur: {valeur}$ | {achats} achats | Inactif depuis {jours_inactif} jours\n"
            f"Actions:\n" + "\n".join(actions)
        )
    except Exception as e:
        return f"Erreur recommandation : {str(e)}"

def run_agent(agent_func, question: str) -> dict:
    """Lance un agent async depuis du code synchrone Streamlit"""
    import nest_asyncio
    import asyncio
    import concurrent.futures

    try:
        # Methode 1 — thread separe (evite le conflit avec la boucle Streamlit)
        with concurrent.futures.ThreadPoolExecutor() as pool:
            future = pool.submit(asyncio.run, agent_func(question))
            return future.result(timeout=60)

    except Exception as e:
        return {
            "agent"  : "erreur",
            "reponse": f"Erreur agent : {str(e)}",
            "tokens" : 0,
            "alertes": [],
            "donnees": []
        }

MOTS_CLES = {
    "politiques": ["leasing","financement","garantie","remise","promotion","prix","politique","condition","vente","contrat","paiement","mensualite","acompte","interet","assurance","taux"],
    "stocks"    : ["stock","piece","disponible","inventaire","livraison","rupture","alerte","quantite","fournisseur","delai","manque"],
    "commandes" : ["commande","suivi","statut","livrer","vendeur","historique","confirme","annule","facture"],
    "clients"   : ["client","fidelite","fidelisation","vip","historique client","recommandation","retention","satisfaction","inactif"]
}

def detecter_agent(question: str) -> str:
    """Detecte le bon agent selon les mots cles de la question"""
    question_lower = question.lower()
    scores = {agent: 0 for agent in MOTS_CLES}
    for agent, mots in MOTS_CLES.items():
        for mot in mots:
            if mot in question_lower:
                scores[agent] += 1
    agent_choisi = max(scores, key=scores.get)
    return "politiques" if scores[agent_choisi] == 0 else agent_choisi

def orchestrer(question: str, agent_force: str = None) -> dict:
    """Route la question vers le bon agent Microsoft Agent Framework"""
    agent      = agent_force if agent_force else detecter_agent(question)
    agents_map = {
        "politiques": agent_politiques,
        "stocks"    : agent_stocks,
        "commandes" : agent_commandes,
        "clients"   : agent_fidelisation,
    }
    return run_agent(agents_map.get(agent, agent_politiques), question)

def index_existe(credential, index_name: str) -> bool:
    try:
        SearchIndexClient(SEARCH_ENDPOINT, credential).get_index(index_name)
        return True
    except:
        return False

def get_embedding(embedding_client, texte: str) -> list:
    return embedding_client.embeddings.create(input=texte, model=EMBEDDING_MODEL).data[0].embedding

def chunker(texte: str, taille=CHUNK_SIZE, overlap=CHUNK_OVERLAP) -> list:
    mots, chunks, i = texte.split(), [], 0
    while i < len(mots):
        chunks.append(" ".join(mots[i:i + taille]))
        i += taille - overlap
    return chunks

def creer_index(credential, index_name: str):
    client = SearchIndexClient(SEARCH_ENDPOINT, credential)
    fields = [
        SimpleField(name="id",      type=SearchFieldDataType.String, key=True),
        SearchableField(name="contenu", type=SearchFieldDataType.String),
        SimpleField(name="fichier", type=SearchFieldDataType.String, filterable=True),
        SimpleField(name="page",    type=SearchFieldDataType.Int32,  filterable=True),
        SimpleField(name="agent",   type=SearchFieldDataType.String, filterable=True),
        SearchField(name="embedding", type=SearchFieldDataType.Collection(SearchFieldDataType.Single), searchable=True, vector_search_dimensions=1536, vector_search_profile_name="profil-vecteur"),
    ]
    vector_search = VectorSearch(
        algorithms=[HnswAlgorithmConfiguration(name="algo-hnsw")],
        profiles=[VectorSearchProfile(name="profil-vecteur", algorithm_configuration_name="algo-hnsw")]
    )
    client.create_or_update_index(SearchIndex(name=index_name, fields=fields, vector_search=vector_search))
    print(f"Index '{index_name}' cree !")

def indexer_pdfs(credential, embedding_client, dossier: str, index_name: str, nom_agent: str):
    if not os.path.exists(dossier):
        os.makedirs(dossier)
        print(f"Dossier '{dossier}' cree — ajoutez vos PDFs dedans")
        return 0, 0
    search_client = SearchClient(SEARCH_ENDPOINT, index_name, credential)
    docs          = []
    fichiers      = [f for f in os.listdir(dossier) if f.endswith(".pdf")]
    if not fichiers:
        print(f"Aucun PDF dans '{dossier}'")
        return 0, 0
    for fichier in fichiers:
        print(f"Traitement : {fichier}")
        reader = PdfReader(f"{dossier}/{fichier}")
        for num_page, page in enumerate(reader.pages):
            texte = page.extract_text()
            if not texte:
                continue
            for chunk in chunker(texte):
                docs.append({"id": str(uuid.uuid4()), "contenu": chunk, "fichier": fichier, "page": num_page + 1, "agent": nom_agent, "embedding": get_embedding(embedding_client, chunk)})
    search_client.upload_documents(docs)
    print(f"{len(docs)} chunks indexes pour '{nom_agent}' !")
    return len(docs), len(fichiers)

def tester_ragas(embedding_client, credential, index_name: str, questions_test: list):
    """Tests RAGAS — questions_test = [{'question': '...', 'ground_truth': '...'}]"""
    try:
        from ragas import evaluate
        from ragas.metrics import faithfulness, answer_relevancy, context_precision, context_recall
        from datasets import Dataset
        questions, answers, contexts, ground_truths = [], [], [], []
        for item in questions_test:
            q_embedding   = get_embedding(embedding_client, item["question"])
            search_client = SearchClient(SEARCH_ENDPOINT, index_name, credential)
            resultats     = list(search_client.search(search_text=item["question"], vector_queries=[{"vector": q_embedding, "fields": "embedding", "k": 5, "kind": "vector"}]))
            questions.append(item["question"])
            answers.append(item.get("reponse", ""))
            contexts.append([r["contenu"] for r in resultats])
            ground_truths.append(item["ground_truth"])
        results = evaluate(Dataset.from_dict({"question": questions, "answer": answers, "contexts": contexts, "ground_truth": ground_truths}), metrics=[faithfulness, answer_relevancy, context_precision, context_recall])
        print("Resultats RAGAS :", results)
        return results
    except ImportError:
        print("RAGAS non installe : pip install ragas datasets")
        return None

def exporter_pdf(titre, contenu, nom_fichier=None):
    """Export PDF — pip install reportlab"""
    try:
        from reportlab.lib.pagesizes import letter
        from reportlab.lib.styles import getSampleStyleSheet, ParagraphStyle
        from reportlab.lib.colors import HexColor
        from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer
        from reportlab.lib.units import inch
        if not nom_fichier:
            nom_fichier = f"exports/rapport_{datetime.now().strftime('%Y%m%d_%H%M%S')}.pdf"
        os.makedirs("exports", exist_ok=True)
        doc     = SimpleDocTemplate(nom_fichier, pagesize=letter)
        styles  = getSampleStyleSheet()
        s_titre = ParagraphStyle('T', parent=styles['Heading1'], fontSize=18, textColor=HexColor('#1a5c1a'), spaceAfter=20)
        s_corps = ParagraphStyle('C', parent=styles['Normal'], fontSize=11, spaceAfter=10, leading=16)
        s_pied  = ParagraphStyle('P', parent=styles['Normal'], fontSize=9, textColor=HexColor('#888888'))
        elems   = [Paragraph(f"{COMPANY_NAME} — {titre}", s_titre), Spacer(1, 0.2*inch), Paragraph(f"Genere le : {datetime.now().strftime('%d/%m/%Y %H:%M')}", s_pied), Spacer(1, 0.3*inch)]
        for ligne in contenu.split('\n'):
            if ligne.strip():
                elems.append(Paragraph(ligne.replace('<','&lt;').replace('>','&gt;'), s_corps))
        doc.build(elems)
        return nom_fichier
    except ImportError:
        return None

defaults = {"messages":[], "nb_questions":0, "notifications":[], "alertes_stocks":[], "agent_actif":"politiques", "auto_detect":True, "index_status":{}}
for k, v in defaults.items():
    if k not in st.session_state:

@st.cache_data(ttl=300)
def verifier_index(_cred):
    return {
        "politiques": index_existe(_cred, INDEX_POLITIQUES),
        "stocks"    : index_existe(_cred, INDEX_STOCKS),
        "commandes" : index_existe(_cred, INDEX_COMMANDES),
        "clients"   : index_existe(_cred, INDEX_CLIENTS),
    }

st.session_state.index_status = verifier_index(credential)

def add_notif(msg, kind="info"):
    st.session_state.notifications.append({"msg": msg, "kind": kind})
    if len(st.session_state.notifications) > 3:
        st.session_state.notifications.pop(0)

def compter_excel(dossier, fichier):
    chemin = f"{dossier}/{fichier}"
    if os.path.exists(chemin):
        try: return len(pd.read_excel(chemin))
        except: return 0
    return 0

with tab_chat:
    for msg in st.session_state.messages:
        if msg["role"] == "user":
            st.markdown(chat_user(msg["content"]), unsafe_allow_html=True)
        else:
            st.markdown(chat_agent(msg["content"], msg.get("agent","politiques")), unsafe_allow_html=True)
    st.markdown("---")
    with st.form("chat_form", clear_on_submit=True):
        user_input = st.text_area("Question", placeholder="Ex: Y a-t-il des ruptures de stock ? / Conditions de leasing ?", height=80, label_visibility="collapsed")
        col1, col2 = st.columns([5,1])
        with col2:
            submitted = st.form_submit_button("Envoyer")
    if submitted and user_input.strip():
        agent = detecter_agent(user_input) if st.session_state.auto_detect else st.session_state.agent_actif
        st.session_state.agent_actif = agent
        st.session_state.messages.append({"role":"user","content":user_input.strip()})
        with st.spinner(f"Agent {agent} (Microsoft SDK) reflechit..."):
            try:
                start    = time.time()
                resultat = orchestrer(user_input.strip(), agent)
                elapsed  = round(time.time() - start, 2)
                if resultat.get("alertes"):
                    st.session_state.alertes_stocks = resultat["alertes"]
                st.session_state.messages.append({"role":"assistant","content":resultat["reponse"],"agent":resultat["agent"]})
                st.session_state.nb_questions += 1
                add_notif(f"Agent {agent} — {elapsed}s", "info")
            except Exception as e:
                add_notif(f"Erreur : {str(e)}", "error")
                st.session_state.messages.pop()
        st.rerun()

with tab_eval:
    st.markdown("### Evaluation RAG")
    col1, col2 = st.columns(2)
    with col1:
        q1  = st.text_input("Question 1")
        gt1 = st.text_input("Reponse attendue 1")
        q2  = st.text_input("Question 2")
        gt2 = st.text_input("Reponse attendue 2")
    with col2:
        agent_eval = st.selectbox("Agent", ["politiques","stocks","commandes","clients"])
        methode    = st.radio("Methode", ["LLM as a Judge","RAGAS"])
    if st.button("Lancer"):
        questions = [q for q in [q1,q2] if q]
        if not questions:
            st.warning("Ajoutez au moins une question")
        else:
            with st.spinner("Evaluation..."):
                if methode == "LLM as a Judge":
                    from backend.config.settings import get_agent_client
                    import asyncio, nest_asyncio
                    nest_asyncio.apply()
                    client = get_agent_client()
                    for question in questions:
                        r      = orchestrer(question, agent_eval)
                        prompt = f'Evalue cette reponse RAG. JSON uniquement: {{"fidelite":{{"score":0,"commentaire":""}},"pertinence":{{"score":0,"commentaire":""}},"completude":{{"score":0,"commentaire":""}},"verdict":"BON/MOYEN/MAUVAIS","suggestion":""}}\nQUESTION: {question}\nREPONSE: {r["reponse"]}'
                        async def eval_llm():
                            a = client.as_agent(name="Judge", instructions="Tu evalues des reponses RAG. Reponds UNIQUEMENT en JSON valide.")
                            return await a.run(prompt)
                        try:
                            resp    = asyncio.get_event_loop().run_until_complete(eval_llm())
                            verdict = json.loads(str(resp).replace("```json","").replace("```","").strip())
                            st.markdown(f"**{question}**")
                            c1,c2,c3 = st.columns(3)
                            c1.metric("Fidelite",   f"{verdict['fidelite']['score']}/10")
                            c2.metric("Pertinence", f"{verdict['pertinence']['score']}/10")
                            c3.metric("Completude", f"{verdict['completude']['score']}/10")
                            st.info(f"Suggestion: {verdict['suggestion']}")
                            st.markdown("---")
                        except Exception as e:
                            st.error(f"Erreur evaluation : {e}")
                else:
                    try:
                        from ragas import evaluate
                        from ragas.metrics import faithfulness, answer_relevancy, context_precision, context_recall
                        from datasets import Dataset
                        answers, contexts = [], []
                        for q in questions:
                            r = orchestrer(q, agent_eval)
                            answers.append(r["reponse"])
                            contexts.append([r["reponse"]])
                        ground_truths = [g for g in [gt1,gt2] if g]
                        results = evaluate(Dataset.from_dict({"question":questions,"answer":answers,"contexts":contexts,"ground_truth":ground_truths[:len(questions)]}), metrics=[faithfulness,answer_relevancy,context_precision,context_recall])
                        c1,c2,c3,c4 = st.columns(4)
                        c1.metric("Faithfulness",     f"{results['faithfulness']:.2f}")
                        c2.metric("Answer Relevancy", f"{results['answer_relevancy']:.2f}")
                        c3.metric("Context Precision",f"{results['context_precision']:.2f}")
                        c4.metric("Context Recall",   f"{results['context_recall']:.2f}")
                    except ImportError:
                        st.error("pip install ragas datasets")

                        sections = [{"titre":f"Rep {i+1}","contenu":m["content"]} for i,m in enumerate(st.session_state.messages) if m["role"]=="assistant"]
                        f = exporter_powerpoint(titre, sections)

                    elif fmt == "Image PNG":
                        f = exporter_image(titre, contenu)

                    st.error(f"Erreur : {str(e)}")
