def index_existe(credential, index_name):
    """Vérifie si l'index existe déjà dans Azure AI Search"""
    try:
        client = SearchIndexClient(SEARCH_ENDPOINT, credential)
        client.get_index(index_name)
        return True
    except:
        return False

def get_embedding(openai_client, texte):
    """Génère un embedding pour un texte"""
    response = openai_client.embeddings.create(
        input=texte,
        model=EMBEDDING_MODEL
    )
    return response.data[0].embedding

def chunker(texte, taille=CHUNK_SIZE, overlap=CHUNK_OVERLAP):
    """Découpe un texte en morceaux avec chevauchement"""
    mots = texte.split()
    chunks = []
    i = 0
    while i < len(mots):
        chunks.append(" ".join(mots[i:i + taille]))
        i += taille - overlap
    return chunks

def creer_index(credential, index_name):
    """Crée un index vectoriel dans Azure AI Search"""
    client = SearchIndexClient(SEARCH_ENDPOINT, credential)
    fields = [
        SimpleField(name="id",      type=SearchFieldDataType.String, key=True),
        SearchableField(name="contenu", type=SearchFieldDataType.String),
        SimpleField(name="fichier", type=SearchFieldDataType.String, filterable=True),
        SimpleField(name="page",    type=SearchFieldDataType.Int32,  filterable=True),
        SimpleField(name="agent",   type=SearchFieldDataType.String, filterable=True),
        SearchField(
            name="embedding",
            type=SearchFieldDataType.Collection(SearchFieldDataType.Single),
            searchable=True,
            vector_search_dimensions=1536,
            vector_search_profile_name="profil-vecteur"
        ),
    ]
    vector_search = VectorSearch(
        algorithms=[HnswAlgorithmConfiguration(name="algo-hnsw")],
        profiles=[VectorSearchProfile(
            name="profil-vecteur",
            algorithm_configuration_name="algo-hnsw"
        )]
    )
    client.create_or_update_index(
        SearchIndex(name=index_name, fields=fields, vector_search=vector_search)
    )
    print(f"✅ Index '{index_name}' créé !")

def indexer_pdfs(credential, openai_client, dossier, index_name, nom_agent):
    """Indexe tous les PDFs d'un dossier dans Azure AI Search"""
    if not os.path.exists(dossier):
        os.makedirs(dossier)
        print(f"📁 Dossier '{dossier}' créé — ajoutez vos PDFs dedans")
        return 0, 0

    search_client = SearchClient(SEARCH_ENDPOINT, index_name, credential)
    docs = []
    fichiers = [f for f in os.listdir(dossier) if f.endswith(".pdf")]

    if not fichiers:
        print(f"⚠️ Aucun PDF trouvé dans '{dossier}'")
        return 0, 0

    for fichier in fichiers:
        print(f"📄 Traitement : {fichier}")
        reader = PdfReader(f"{dossier}/{fichier}")
        for num_page, page in enumerate(reader.pages):
            texte = page.extract_text()
            if not texte:
                continue
            for chunk in chunker(texte):
                docs.append({
                    "id"       : str(uuid.uuid4()),
                    "contenu"  : chunk,
                    "fichier"  : fichier,
                    "page"     : num_page + 1,
                    "agent"    : nom_agent,
                    "embedding": get_embedding(openai_client, chunk)
                })

    search_client.upload_documents(docs)
    print(f"✅ {len(docs)} chunks indexés pour '{nom_agent}' !")
    return len(docs), len(fichiers)

from azure.search.documents import SearchClient
from backend.config.settings import SEARCH_ENDPOINT, EMBEDDING_MODEL, TOP_K_RESULTS

def get_embedding(openai_client, texte):
    """Transforme un texte en vecteur numérique"""
    response = openai_client.embeddings.create(
        input=texte,
        model=EMBEDDING_MODEL
    )
    return response.data[0].embedding

def rag_query(credential, openai_client, question, index_name, nb_resultats=TOP_K_RESULTS):
    """Cherche les passages pertinents dans Azure AI Search"""
    question_embedding = get_embedding(openai_client, question)
    search_client = SearchClient(SEARCH_ENDPOINT, index_name, credential)
    resultats = list(search_client.search(
        search_text=question,
        vector_queries=[{
            "vector": question_embedding,
            "fields": "embedding",
            "k"     : nb_resultats,
            "kind"  : "vector"
        }]
    ))
    return [
        {"contenu": r["contenu"], "fichier": r["fichier"], "page": r["page"]}
        for r in resultats
    ]

def appel_gpt(openai_client, system_prompt, question, contexte, deployment):
    """Appelle GPT avec le contexte trouvé par le RAG"""
    response = openai_client.chat.completions.create(
        model=deployment,
        messages=[
            {"role": "system", "content": system_prompt},
            {"role": "user",   "content": f"Documents:\n{contexte}\n\nQuestion: {question}"}
        ],
        max_tokens=800,
        temperature=0.3,
    )
    return response.choices[0].message.content, response.usage.total_tokens

def agent_stocks(credential, openai_client, question):
    """Agent : Stocks et pièces — génère des alertes automatiques"""
    import os
    alertes = []
    donnees = []

    fichier_stocks = f"{DOSSIER_STOCKS}/stocks.xlsx"
    if os.path.exists(fichier_stocks):
        df  = pd.read_excel(fichier_stocks)
        col = COLONNES_STOCKS
        for _, row in df.iterrows():
            try:
                quantite    = int(row[col["quantite"]])
                nom_piece   = row[col["nom"]]
                fournisseur = row.get(col["fournisseur"], "N/A")

                if quantite < SEUIL_STOCK_CRITIQUE:
                    alertes.append({"type": "CRITIQUE",  "message": f"🔴 Stock CRITIQUE : {nom_piece} — {quantite} unités", "piece": nom_piece, "quantite": quantite, "fournisseur": fournisseur})
                elif quantite < SEUIL_STOCK_BAS:
                    alertes.append({"type": "ATTENTION", "message": f"🟡 Stock bas : {nom_piece} — {quantite} unités",     "piece": nom_piece, "quantite": quantite, "fournisseur": fournisseur})

                date_livraison = row.get(col["date_livraison"])
                if pd.notna(date_livraison):
                    if isinstance(date_livraison, str):
                        date_livraison = datetime.strptime(date_livraison, "%Y-%m-%d")
                    if date_livraison < datetime.now() and quantite < int(row[col["seuil_minimum"]]):
                        alertes.append({"type": "RETARD", "message": f"⏰ Retard livraison : {nom_piece} — prévu le {date_livraison.strftime('%d/%m/%Y')}", "piece": nom_piece, "quantite": quantite, "fournisseur": fournisseur})

                donnees.append({"piece": nom_piece, "quantite": quantite, "fournisseur": fournisseur})
            except Exception:
                continue

    system_prompt = """
    Tu es un gestionnaire de stocks automobile chez AutoPlus.
    Réponds aux questions sur les stocks, pièces disponibles,
    délais de livraison et alertes de rupture.
    Sois précis avec les chiffres et les dates.

    reponse, tokens  = appel_gpt(openai_client, system_prompt, question, contexte_complet, AZURE_OPENAI_DEPLOYMENT)
    return reponse, alertes, donnees, tokens

from backend.agents.agents import agent_politiques, agent_stocks, agent_commandes, agent_fidelisation

# Mots clés par agent — modifiez selon vos besoins
MOTS_CLES_AGENTS = {
    "politiques": ["leasing","financement","garantie","remise","promotion","prix","politique","condition","vente","contrat","paiement","mensualité","acompte","intérêt","assurance","taux"],
    "stocks"    : ["stock","pièce","disponible","inventaire","livraison","rupture","alerte","quantité","fournisseur","délai"],
    "commandes" : ["commande","suivi","statut","livrer","vendeur","historique","confirmé","annulé","facture"],
    "clients"   : ["client","fidélité","fidélisation","vip","historique client","recommandation","promotion client","rétention","satisfaction"]
}

def detecter_agent(question):
    """Détecte automatiquement le bon agent selon la question"""
    question_lower = question.lower()
    scores = {"politiques": 0, "stocks": 0, "commandes": 0, "clients": 0}
    for agent, mots in MOTS_CLES_AGENTS.items():
        for mot in mots:
            if mot in question_lower:
                scores[agent] += 1
    agent_choisi = max(scores, key=scores.get)
    return "politiques" if scores[agent_choisi] == 0 else agent_choisi

def orchestrer(credential, openai_client, question, agent_force=None):
    """Route la question vers le bon agent"""
    agent    = agent_force if agent_force else detecter_agent(question)
    resultat = {"agent": agent, "reponse": "", "sources": [], "tokens": 0, "alertes": [], "donnees": []}

    if agent == "politiques":
        reponse, sources, tokens = agent_politiques(credential, openai_client, question)
        resultat.update({"reponse": reponse, "sources": sources, "tokens": tokens})
    elif agent == "stocks":
        reponse, alertes, donnees, tokens = agent_stocks(credential, openai_client, question)
        resultat.update({"reponse": reponse, "alertes": alertes, "donnees": donnees, "tokens": tokens})
    elif agent == "commandes":
        reponse, donnees, tokens = agent_commandes(credential, openai_client, question)
        resultat.update({"reponse": reponse, "donnees": donnees, "tokens": tokens})
    elif agent == "clients":
        reponse, donnees, tokens = agent_fidelisation(credential, openai_client, question)
        resultat.update({"reponse": reponse, "donnees": donnees, "tokens": tokens})

    return resultat

        style_titre = ParagraphStyle('Titre', parent=styles['Heading1'], fontSize=18, textColor=HexColor('#1a5c1a'), spaceAfter=20)
        style_corps = ParagraphStyle('Corps', parent=styles['Normal'], fontSize=11, spaceAfter=10, leading=16)
        style_pied  = ParagraphStyle('Pied',  parent=styles['Normal'], fontSize=9,  textColor=HexColor('#888888'))

        elements = []
        elements.append(Paragraph(f"{COMPANY_NAME} — {titre}", style_titre))
        elements.append(Spacer(1, 0.2*inch))
        elements.append(Paragraph(f"Généré le : {datetime.now().strftime('%d/%m/%Y %H:%M')}", style_pied))
        elements.append(Spacer(1, 0.3*inch))
        for ligne in contenu.split('\n'):
            if ligne.strip():
                elements.append(Paragraph(ligne.replace('<','&lt;').replace('>','&gt;'), style_corps))
        doc.build(elements)
        return nom_fichier
    except ImportError:
        return None

def exporter_powerpoint(titre, sections, nom_fichier=None):
    """Exporte en PowerPoint — nécessite : pip install python-pptx"""
    try:
        from pptx import Presentation
        from pptx.util import Pt
        from pptx.dml.color import RGBColor

        if not nom_fichier:
            nom_fichier = f"exports/rapport_{datetime.now().strftime('%Y%m%d_%H%M%S')}.pptx"
        os.makedirs("exports", exist_ok=True)

        prs   = Presentation()
        slide = prs.slides.add_slide(prs.slide_layouts[0])
        slide.shapes.title.text = COMPANY_NAME
        slide.placeholders[1].text = titre
        slide.shapes.title.text_frame.paragraphs[0].runs[0].font.color.rgb = RGBColor(0x1a, 0x5c, 0x1a)

        for section in sections:
            slide = prs.slides.add_slide(prs.slide_layouts[1])
            slide.shapes.title.text = section["titre"]
            slide.placeholders[1].text_frame.text = section["contenu"][:500]

        slide = prs.slides.add_slide(prs.slide_layouts[0])
        slide.shapes.title.text = "Merci"
        slide.placeholders[1].text = f"Généré par {COMPANY_NAME} IA — {datetime.now().strftime('%d/%m/%Y')}"
        prs.save(nom_fichier)
        return nom_fichier
    except ImportError:
        return None

def exporter_image(titre, contenu, nom_fichier=None):
    """Exporte en image PNG — nécessite : pip install Pillow"""
    try:
        from PIL import Image, ImageDraw

        if not nom_fichier:
            nom_fichier = f"exports/rapport_{datetime.now().strftime('%Y%m%d_%H%M%S')}.png"
        os.makedirs("exports", exist_ok=True)

        lignes  = contenu.split('\n')
        hauteur = max(400, 120 + len(lignes) * 22)
        img     = Image.new('RGB', (900, hauteur), color='#0a1f0a')
        draw    = ImageDraw.Draw(img)
        draw.rectangle([0, 0, 900, 70], fill='#122612')
        draw.text((20, 15), f"{COMPANY_NAME} — {titre}", fill='#39ff14')
        draw.text((20, 45), f"Généré le {datetime.now().strftime('%d/%m/%Y %H:%M')}", fill='#5ec45e')
        y = 90
        for ligne in lignes:
            if ligne.strip():
                draw.text((20, y), ligne[:100], fill='#c2f0c2')
                y += 22
        img.save(nom_fichier)
        return nom_fichier
    except ImportError:
        return None

    if st.button("🧪 Lancer l'évaluation"):
        questions     = [q for q in [q1,q2] if q]
        ground_truths = [g for g in [gt1,gt2] if g]
        if not questions:
            st.warning("Ajoutez au moins une question")
        else:
            with st.spinner("Évaluation..."):
                if methode == "LLM as a Judge":
                    for question in questions:
                        r = orchestrer(credential, openai_client, question, agent_eval)
                        prompt = f"""Évalue cette réponse RAG. Retourne UNIQUEMENT un JSON :
{{"fidelite":{{"score":0,"commentaire":""}},"pertinence":{{"score":0,"commentaire":""}},"completude":{{"score":0,"commentaire":""}},"verdict":"BON/MOYEN/MAUVAIS","suggestion":""}}
QUESTION: {question}
RÉPONSE: {r['reponse']}"""
                        resp    = openai_client.chat.completions.create(model="gpt-4-5", messages=[{"role":"user","content":prompt}], max_tokens=400, temperature=0)
                        verdict = json.loads(resp.choices[0].message.content.replace("```json","").replace("```","").strip())
                        st.markdown(f"**{question}**")
                        c1,c2,c3 = st.columns(3)
                        c1.metric("Fidélité",   f"{verdict['fidelite']['score']}/10")
                        c2.metric("Pertinence", f"{verdict['pertinence']['score']}/10")
                        c3.metric("Complétude", f"{verdict['completude']['score']}/10")
                        st.info(f"💡 {verdict['suggestion']}")
                        st.markdown("---")
                else:
                    try:
                        from ragas import evaluate
                        from ragas.metrics import faithfulness, answer_relevancy, context_precision, context_recall
                        from datasets import Dataset
                        answers  = []
                        contexts = []
                        for q in questions:
                            r = orchestrer(credential, openai_client, q, agent_eval)
                            answers.append(r["reponse"])
                            contexts.append([s["contenu"] for s in r["sources"]] if r["sources"] else [""])
                        results = evaluate(Dataset.from_dict({"question":questions,"answer":answers,"contexts":contexts,"ground_truth":ground_truths[:len(questions)]}), metrics=[faithfulness,answer_relevancy,context_precision,context_recall])
                        c1,c2,c3,c4 = st.columns(4)
                        c1.metric("Faithfulness",     f"{results['faithfulness']:.2f}")
                        c2.metric("Answer Relevancy", f"{results['answer_relevancy']:.2f}")
                        c3.metric("Context Precision",f"{results['context_precision']:.2f}")
                        c4.metric("Context Recall",   f"{results['context_recall']:.2f}")
                    except ImportError:
                        st.error("pip install ragas datasets")
